docker build --no-cache -t php82-nginx:1.0 .


docker run -p 8080:80 -v $(pwd):/var/www/html php82-nginx:1.0

version: "3.8"

services:
  app:
    image: php82-nginx:1.0
    ports:
      - "8080:80"
    volumes:
      - ./src:/var/www/html
      - ./nginx.conf:/etc/nginx/conf.d/default.conf
      - ./php.ini:/usr/local/etc/php/php.ini

composer install
php artisan key:generate
php artisan storage:link
